#### 不用循环实现数组填充

```
var arr = [];
var add = (()=>{
        let i = 1;
        return function(){
            return i++;
        }
    })()
while(arr.length<100){
    arr.push(add());
}
```

```
var arr = [];
let num = {
    valueOf: (()=> {
        let i = 1;
        return ()=>i++;
    })()
}
while(arr.length<100){
    arr.push(num+0);
}
```
